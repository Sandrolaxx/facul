#include <stdio.h>
#include<string.h>

//Struct

struct Trabalim{
    double valor1;
    double valor2;
    int operacao;
    int resultado;
    char matriz[4][10];
};

//Função da Soma
double soma(double n1,double n2){
	printf("O resultado da soma é %.2lf\n",(n1+n2));
}

//Procedimento da subtração
double subtracao(double n1,double n2){

	double Subtracao;
  	  Subtracao = n1 - n2;
	return  Subtracao;
}

//Função da Divisão
double divisao(double n1,double n2){
	printf("O resultado da divisao é %.2lf\n",(n1/n2));
}

//Procedimento da multiplicação
double multiplicacao(double n1,double n2){

     double Mult;
  	  Mult = n1 * n2;
	return Mult;
}

 switch(calculadora[i].operacao){
